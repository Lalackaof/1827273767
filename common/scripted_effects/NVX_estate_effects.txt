fourth_estate_islamist_increase = {
	custom_effect_tooltip = TT_fourth_estate_islamist_increase
	hidden_effect = {
		if = {
			limit = { has_idea = fourth_estate_islamist5 }
			add_popularity = {
				ideology = islamist
				popularity = 0.05
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist4 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist4
				add_idea = fourth_estate_islamist5
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist3 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist3
				add_idea = fourth_estate_islamist4
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist2 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist2
				add_idea = fourth_estate_islamist3
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist1 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist1
				add_idea = fourth_estate_islamist2
			}
		}
		if = {
			limit = {
				has_islamist_fourth_estate = no
				OR = {
					has_idea = fifth_estate_islamist4
					has_idea = fifth_estate_islamist5
				}
			}
			add_ideas = fourth_estate_islamist2
			remove_ideas = fifth_estate_islamist4
			remove_ideas = fifth_estate_islamist5
		}
		if = {
			limit = {
				has_islamist_fourth_estate = no
				OR = {
					has_idea = fifth_estate_islamist1
					has_idea = fifth_estate_islamist2
					has_idea = fifth_estate_islamist3
				}
			}
			add_ideas = fourth_estate_islamist1
			add_popularity = {
				ideology = islamist
				popularity = 0.02
			}
			remove_ideas = fifth_estate_islamist1
			remove_ideas = fifth_estate_islamist2
			remove_ideas = fifth_estate_islamist3
		}
		if = {
			limit = { has_islamist_fourth_estate = no }
			add_ideas = fourth_estate_islamist1
		}
	}
}

fifth_estate_aberrant_increase_effect = {
	if = {
		limit = {
			OR = {
				has_government = islamist
				has_government = fascist
				has_government = nationalist
				has_government = monarchist
			}
		}
		fifth_estate_aberrant_shift_oppose_right_radicals = yes
	}
	if = {
		limit = {
			OR = {
				has_government = reactionary
				has_government = conservative
				has_government = market_liberal
			}
		}
		fifth_estate_aberrant_shift_oppose_right = yes
	}
	if = {
		limit = {
			OR = {
				has_government = social_liberal
				has_government = social_democrat
				has_government = progressive
			}
		}
		fifth_estate_aberrant_shift_oppose_left = yes
	}
	if = {
		limit = {
			OR = {
				has_government = democratic_socialist
				has_government = communist
			}
		}
		fifth_estate_aberrant_shift_oppose_left_radicals = yes
	}
}
