country_event = { #Fifth estate natural increase islamist
	id = nvx_fifth_estates.1
	hidden = yes
	
	trigger = {
		NOT = { has_global_flag = sandbox_disable_natural_estate_drift }
		has_islamist_fourth_estate = no
		OR = {
			AND = {
				islamist > 0.049
				has_islamist_fifth_estate = no
			}
			AND = {
				islamist > 0.099
				NOT = {
					OR = {
						has_idea = fifth_estate_islamist2
						has_idea = fifth_estate_islamist3
						has_idea = fifth_estate_islamist4
						has_idea = fifth_estate_islamist5
					}
				}
			}
			AND = {
				islamist > 0.149
				NOT = {
					OR = {
						has_idea = fifth_estate_islamist3
						has_idea = fifth_estate_islamist4
						has_idea = fifth_estate_islamist5
					}
				}
			}
			AND = {
				islamist > 0.199
				NOT = {
					OR = {
						has_idea = fifth_estate_islamist4
						has_idea = fifth_estate_islamist5
					}
				}
			}
			AND = {
				islamist > 0.249
				NOT = {
					has_idea = fifth_estate_islamist5
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 250
		modifier = { factor = 0.5 has_stability < 0.2 }
		modifier = { factor = 0.5 has_stability < 0.4 }
		modifier = { factor = 1.5 has_stability > 0.74 }
		modifier = { factor = 0.65 has_idea = national_development_developed }
		modifier = { factor = 1.35 has_idea = national_development_undeveloped }
		modifier = { factor = 1.35 has_idea = civil_liberties_totalitarian OR = { has_idea = public_safety_central_law_office has_idea = public_safety_secret_police } }
		modifier = { factor = 1.3 has_idea = civil_liberties_totalitarian has_idea = public_safety_subsidized_agencies }
		modifier = { factor = 1.25 has_idea = civil_liberties_restricted OR = { has_idea = public_safety_central_law_office has_idea = public_safety_secret_police } }
		modifier = { factor = 1.2 has_idea = civil_liberties_restricted has_idea = public_safety_subsidized_agencies }
		modifier = { factor = 1.15 has_idea = civil_liberties_limited OR = { has_idea = public_safety_central_law_office has_idea = public_safety_secret_police } }
		modifier = { factor = 1.1 has_idea = civil_liberties_limited has_idea = public_safety_subsidized_agencies }
		modifier = { factor = 0.3 has_idea = fifth_estate_islamist1 islamist > 0.149 }
		modifier = { factor = 0.3 has_idea = fifth_estate_islamist2 islamist > 0.199 }
		modifier = { factor = 0.3 has_idea = fifth_estate_islamist3 islamist > 0.249 }
		modifier = { factor = 0.3 has_idea = fifth_estate_islamist4 islamist > 0.299 }
	}
	
	immediate = {
		set_country_flag = NVX_flag_fifth_estate_islamist
		if = {
			limit = { NOT = { has_country_flag = NVX_estate_natural_changes_hide } }
