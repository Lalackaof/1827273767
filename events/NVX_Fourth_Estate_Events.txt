country_event = {#Clear 5th estates if alongside 4th estates
	id = nvx_fourth_estates.0
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				has_islamist_fourth_estate = yes
				has_islamist_fifth_estate = yes
			}
			AND = {
				has_fascist_fourth_estate = yes
				has_fascist_fifth_estate = yes
			}
			AND = {
				has_nationalist_fourth_estate = yes
				has_nationalist_fifth_estate = yes
			}
			AND = {
				has_monarchist_fourth_estate = yes
				has_monarchist_fifth_estate = yes
			}
			AND = {
				has_reactionary_fourth_estate = yes
				has_reactionary_fifth_estate = yes
			}
			AND = {
				has_conservative_fourth_estate = yes
				has_conservative_fifth_estate = yes
			}
			AND = {
				has_market_liberal_fourth_estate = yes
				has_market_liberal_fifth_estate = yes
			}
			AND = {
				has_social_liberal_fourth_estate = yes
				has_social_liberal_fifth_estate = yes
			}

				has_social_democrat_fourth_estate = yes

				has_democratic_socialist_fourth_estate = yes
				has_democratic_socialist_fifth_estate = yes
			}
			AND = {
				has_communist_fourth_estate = yes
				has_communist_fifth_estate = yes
			}
		}
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		if = {
			limit = {
				has_islamist_fourth_estate = yes
				has_islamist_fifth_estate = yes

		}
		if = {
			limit = {
				has_fascist_fourth_estate = yes
				has_fascist_fifth_estate = yes
			}
			fifth_estate_fascist_decrease = yes
		}
		if = {
			limit = {
				has_nationalist_fourth_estate = yes
				has_nationalist_fifth_estate = yes
			}
			fifth_estate_nationalist_decrease = yes
		}
		if = {
			limit = {
				has_monarchist_fourth_estate = yes
				has_monarchist_fifth_estate = yes
			}
			fifth_estate_monarchist_decrease = yes
		}
		if = {
			limit = {
				has_reactionary_fourth_estate = yes
				has_reactionary_fifth_estate = yes
			}
			fifth_estate_reactionary_decrease = yes
		}
		if = {
			limit = {
				has_conservative_fourth_estate = yes
				has_conservative_fifth_estate = yes
			}
			fifth_estate_conservative_decrease = yes
		}
		if = {
			limit = {
				has_market_liberal_fourth_estate = yes
				has_market_liberal_fifth_estate = yes
			}
			fifth_estate_market_liberal_decrease = yes
		}

				has_social_liberal_fourth_estate = yes

country_event = { #Fourth estate natural increase islamist
	id = nvx_fourth_estates.1
	hidden = yes
	
	trigger = {
		NOT = { has_global_flag = sandbox_disable_natural_estate_drift }
		NOT = { has_idea = civil_liberties_totalitarian }
		NOT = {
			AND = {
				has_government = islamist
				NVX_has_government_system_authoritarian = yes
			}
		}
		OR = {
			AND = {
				islamist > 0.199
				has_islamist_fourth_estate = no
			}
			AND = {
				islamist > 0.299
				NOT = {
					OR = {
						has_idea = fourth_estate_islamist2
						has_idea = fourth_estate_islamist3
						has_idea = fourth_estate_islamist4
						has_idea = fourth_estate_islamist5
					}
				}
			}
			AND = {
				islamist > 0.399
				NOT = {
					OR = {
						has_idea = fourth_estate_islamist3
						has_idea = fourth_estate_islamist4
						has_idea = fourth_estate_islamist5
					}
				}
			}
			AND = {
				islamist > 0.499
				NOT = {
					OR = {
						has_idea = fourth_estate_islamist4
						has_idea = fourth_estate_islamist5
					}
				}
			}
			AND = {
				islamist > 0.599
				NOT = {
					has_idea = fourth_estate_islamist5
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 420
		modifier = { factor = 1.5 has_idea = civil_liberties_restricted }
		modifier = { factor = 1.25 has_idea = civil_liberties_limited }
		modifier = { factor = 0.5 OR = { has_idea = fifth_estate_islamist4 has_idea = fifth_estate_islamist5 } }
		modifier = { factor = 0.3 has_idea = fourth_estate_islamist1 islamist > 0.399 }
		modifier = { factor = 0.3 has_idea = fourth_estate_islamist2 islamist > 0.499 }
		modifier = { factor = 0.3 has_idea = fourth_estate_islamist3 islamist > 0.599 }
		modifier = { factor = 0.3 has_idea = fourth_estate_islamist4 islamist > 0.699 }
	}
	
	immediate = {
		set_country_flag = NVX_flag_fourth_estate_islamist
		if = {
			limit = { NOT = { has_country_flag = NVX_estate_natural_changes_hide } }
			country_event = { id = nvx_fourth_estates.25 }
		}
		if = {
			limit = { has_country_flag = NVX_estate_natural_changes_hide }
			fourth_estate_targeted_total_increase = yes
		}
	}
}
