#Alberto Fujimori's exile
country_event = {
	id = peru.1
	title = peru.1.t
	desc = peru.1.desc
	picture = GFX_report_event_JAP_fujimori_airplane
	
	trigger = {
		original_tag = PRU
		date > 2000.06.01
	}
	
	mean_time_to_happen = {
		days = 180
	}
	
	immediate = {
		hidden_effect = {
			add_opinion_modifier = {
				target = JAP
				modifier = JAP_accepting_fujimori_s_exile
			}
		}
	}
	
	fire_only_once = yes
	
	immediate = {
		hidden_effect = { 
			JAP = { country_event = { id = peru.2 } } 
			news_event = { id = japan_news.15 days = 3 }
		}
		
		retire_character = PRU_alberto_fujimori
	}
	
	option = {
		name = peru.1.a
		set_politics = {
			ruling_party = market_liberal
			elections_allowed = yes
		}
		add_popularity = {
			ideology = reactionary
			popularity = -0.5
		}
	}
}
